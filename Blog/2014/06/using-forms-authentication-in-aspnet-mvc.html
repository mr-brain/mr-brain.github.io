    Using Forms Authentication in ASP.NET MVC

    Posted on 2014/06/25 22:12:59

    In Web.config

    <configuration>
  <system.web>
    <authentication mode="Forms">
      <forms defaultUrl="~/Home/Index/" loginUrl="~/Home/Login/" timeout="30" ></forms>
    </authentication>
    <authorization>
      <deny users="?"></deny>
    </authorization>
  </system.web>
  <location path="Scripts">
    <system.web>
	  <authorization>
	    <allow users="*"></allow>
	  </authorization>
	</system.web>
  </location>
</configuration>
